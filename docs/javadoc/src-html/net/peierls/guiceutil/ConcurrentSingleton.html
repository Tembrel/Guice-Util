001package net.peierls.guiceutil;
002
003import java.lang.annotation.Documented;
004import java.lang.annotation.ElementType;
005import java.lang.annotation.Retention;
006import static java.lang.annotation.RetentionPolicy.RUNTIME;
007import java.lang.annotation.Target;
008
009import com.google.inject.Injector;
010import javax.inject.Scope;
011
012/**
013 * Apply this to implementation classes when you want only one instance
014 * (per {@link Injector}) to be reused for all injections for that binding,
015 * and you want to eagerly create the instance concurrently in the background
016 * with other singletons in this scope.
017 */
018@Documented
019@Target({ ElementType.TYPE, ElementType.METHOD })
020@Retention(RUNTIME)
021@Scope
022public @interface ConcurrentSingleton {}
